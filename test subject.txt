local ALLOWED_KEYS = {"oskiuwuporn", "dustydepot01", "zemi"}  -- list of valid keys

local Players = game:GetService("Players")
local player = Players.LocalPlayer

-- Create the key prompt GUI

-- Main frame
local frame = Instance.new("Frame")

frame.AnchorPoint = Vector2.new(0.5, 0.5)
frame.Parent = gui

-- Rounded corners
local uICorner = Instance.new("UICorner")
uICorner.CornerRadius = UDim.new(0, 12)
uICorner.Parent = frame

-- Shadow effect
local uiStroke = Instance.new("UIStroke")
uiStroke.Color = Color3.fromRGB(60, 60, 70)
uiStroke.Thickness = 1
uiStroke.Parent = frame

-- Title label
local label = Instance.new("TextLabel")
label.Size = UDim2.new(1, -40, 0, 50)
label.Position = UDim2.new(0, 20, 0, 0)
label.BackgroundTransparency = 1
label.TextColor3 = Color3.fromRGB(230, 230, 230)
label.Text = "Enter Key to Unlock Pumpkin Hub"
label.Font = Enum.Font.GothamBold
label.TextSize = 15
label.TextXAlignment = Enum.TextXAlignment.Left
label.Parent = frame

-- Close (X) button
local closeButton = Instance.new("TextButton")
closeButton.Size = UDim2.new(0, 30, 0, 30)
closeButton.Position = UDim2.new(1, -35, 0, 10)
closeButton.Text = "✕"
closeButton.Font = Enum.Font.GothamBold
closeButton.TextSize = 20
closeButton.TextColor3 = Color3.fromRGB(255, 100, 100)
closeButton.BackgroundTransparency = 1
closeButton.Parent = frame

closeButton.MouseEnter:Connect(function()
    closeButton.TextColor3 = Color3.fromRGB(255, 50, 50)
end)
closeButton.MouseLeave:Connect(function()
    closeButton.TextColor3 = Color3.fromRGB(255, 100, 100)
end)

closeButton.MouseButton1Click:Connect(function()
    gui:Destroy()
end)

-- Key textbox
local textBox = Instance.new("TextBox")
textBox.Size = UDim2.new(1, -40, 0, 40)
textBox.Position = UDim2.new(0, 20, 0, 60)
textBox.PlaceholderText = "Enter key here"
textBox.Text = ""
textBox.BackgroundColor3 = Color3.fromRGB(45, 45, 50)
textBox.TextColor3 = Color3.fromRGB(230, 230, 230)
textBox.Font = Enum.Font.Gotham
textBox.TextSize = 18
textBox.ClearTextOnFocus = false
textBox.Parent = frame

local textBoxCorner = Instance.new("UICorner")
textBoxCorner.CornerRadius = UDim.new(0, 8)
textBoxCorner.Parent = textBox

-- Submit button
local submit = Instance.new("TextButton")
submit.Size = UDim2.new(1, -40, 0, 40)
submit.Position = UDim2.new(0, 20, 0, 115)
submit.Text = "Submit"
submit.Font = Enum.Font.GothamBold
submit.TextSize = 18
submit.TextColor3 = Color3.fromRGB(255, 255, 255)
submit.BackgroundColor3 = Color3.fromRGB(70, 120, 240)
submit.Parent = frame

local submitCorner = Instance.new("UICorner")
submitCorner.CornerRadius = UDim.new(0, 8)
submitCorner.Parent = submit

submit.MouseEnter:Connect(function()
    submit.BackgroundColor3 = Color3.fromRGB(90, 140, 255)
end)
submit.MouseLeave:Connect(function()
    submit.BackgroundColor3 = Color3.fromRGB(70, 120, 240)
end)

-- Animate frame into screen
frame:TweenPosition(UDim2.new(0.5, 0, 0.5, 0), Enum.EasingDirection.Out, Enum.EasingStyle.Back, 0.7, true)

-- Draggable frame
local dragging = false
local dragInput, mousePos, framePos

frame.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        dragging = true
        mousePos = input.Position
        framePos = frame.Position

        input.Changed:Connect(function()
            if input.UserInputState == Enum.UserInputState.End then
                dragging = false
            end
        end)
    end
end)

frame.InputChanged:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseMovement then
        dragInput = input
    end
end)

game:GetService("UserInputService").InputChanged:Connect(function(input)
    if input == dragInput and dragging then
        local delta = input.Position - mousePos
        frame.Position = UDim2.new(framePos.X.Scale, framePos.X.Offset + delta.X,
                                   framePos.Y.Scale, framePos.Y.Offset + delta.Y)
    end
end)

-- Key check logic
local unlocked = false
submit.MouseButton1Click:Connect(function()
    local inputKey = textBox.Text
    local valid = false
    for _, key in pairs(ALLOWED_KEYS) do
        if inputKey == key then
            valid = true
            break
        end
    end

    if valid then
        unlocked = true
        gui:Destroy()
        print("[Hub] Key accepted, loading script…")
    else
        textBox.Text = ""
        textBox.PlaceholderText = "Incorrect key!"
        textBox.TextColor3 = Color3.fromRGB(255, 100, 100)
    end
end)

-- Block rest of the script until unlocked
repeat task.wait() until unlocked

Tabs.VRF:AddButton({
    Title = "Destroy Shields (others only)",
    Description = "Removes any part named 'Shield' from all OTHER players' characters",
    Callback = function()
        local Players = game:GetService("Players")
        local localPlayer = Players.LocalPlayer

        for _, plr in ipairs(Players:GetPlayers()) do
            if plr ~= localPlayer then -- skip yourself
                local char = plr.Character
                if char then
                    for _, child in ipairs(char:GetDescendants()) do
                        if child.Name == "Shield" then
                            -- optional type check:
                            -- if child:IsA("Accessory") or child:IsA("Part") then
                            child:Destroy()
                        end
                    end
                end
            end
        end
    end
})
